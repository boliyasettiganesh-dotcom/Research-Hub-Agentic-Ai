import React, { useState, useEffect } from "react";
import { base44 } from "@/api/base44Client";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
} from "@/components/ui/dialog";
import { Input } from "@/components/ui/input";
import { Textarea } from "@/components/ui/textarea";
import { Button } from "@/components/ui/button";
import { Label } from "@/components/ui/label";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { Upload, Loader2 } from "lucide-react";

export default function PaperFormDialog({ open, onOpenChange, onSubmit, paper, workspaces, defaultWorkspaceId }) {
  const [form, setForm] = useState({
    title: "", authors: "", abstract: "", year: "", journal: "", doi: "",
    workspace_id: "", file_url: "", key_findings: "", methodology: "", tags: [],
  });
  const [uploading, setUploading] = useState(false);
  const [tagInput, setTagInput] = useState("");

  useEffect(() => {
    if (paper) {
      setForm({
        title: paper.title || "", authors: paper.authors || "", abstract: paper.abstract || "",
        year: paper.year || "", journal: paper.journal || "", doi: paper.doi || "",
        workspace_id: paper.workspace_id || "", file_url: paper.file_url || "",
        key_findings: paper.key_findings || "", methodology: paper.methodology || "",
        tags: paper.tags || [],
      });
    } else {
      setForm({
        title: "", authors: "", abstract: "", year: "", journal: "", doi: "",
        workspace_id: defaultWorkspaceId || "", file_url: "", key_findings: "", methodology: "", tags: [],
      });
    }
  }, [paper, open, defaultWorkspaceId]);

  const handleFileUpload = async (e) => {
    const file = e.target.files[0];
    if (!file) return;
    setUploading(true);
    const { file_url } = await base44.integrations.Core.UploadFile({ file });
    setForm(p => ({ ...p, file_url }));
    setUploading(false);
  };

  const handleAddTag = (e) => {
    if (e.key === "Enter" && tagInput.trim()) {
      e.preventDefault();
      setForm(p => ({ ...p, tags: [...p.tags, tagInput.trim()] }));
      setTagInput("");
    }
  };

  const handleSubmit = () => {
    const data = { ...form };
    if (data.year) data.year = Number(data.year);
    else delete data.year;
    onSubmit(data);
  };

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="sm:max-w-lg max-h-[90vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle>{paper ? "Edit Paper" : "Add Paper"}</DialogTitle>
        </DialogHeader>
        <div className="space-y-3">
          <div>
            <Label>Title *</Label>
            <Input value={form.title} onChange={e => setForm(p => ({ ...p, title: e.target.value }))} placeholder="Paper title" />
          </div>
          <div>
            <Label>Authors</Label>
            <Input value={form.authors} onChange={e => setForm(p => ({ ...p, authors: e.target.value }))} placeholder="Author 1, Author 2..." />
          </div>
          <div>
            <Label>Workspace *</Label>
            <Select value={form.workspace_id} onValueChange={v => setForm(p => ({ ...p, workspace_id: v }))}>
              <SelectTrigger><SelectValue placeholder="Select workspace" /></SelectTrigger>
              <SelectContent>
                {workspaces?.map(ws => <SelectItem key={ws.id} value={ws.id}>{ws.name}</SelectItem>)}
              </SelectContent>
            </Select>
          </div>
          <div>
            <Label>Abstract</Label>
            <Textarea value={form.abstract} onChange={e => setForm(p => ({ ...p, abstract: e.target.value }))} placeholder="Paper abstract..." rows={3} />
          </div>
          <div className="grid grid-cols-2 gap-3">
            <div>
              <Label>Year</Label>
              <Input type="number" value={form.year} onChange={e => setForm(p => ({ ...p, year: e.target.value }))} placeholder="2024" />
            </div>
            <div>
              <Label>Journal</Label>
              <Input value={form.journal} onChange={e => setForm(p => ({ ...p, journal: e.target.value }))} placeholder="Nature, IEEE..." />
            </div>
          </div>
          <div>
            <Label>DOI</Label>
            <Input value={form.doi} onChange={e => setForm(p => ({ ...p, doi: e.target.value }))} placeholder="10.1000/xyz123" />
          </div>
          <div>
            <Label>Upload File</Label>
            <div className="flex items-center gap-2">
              <label className="flex-1">
                <div className="border border-dashed border-border rounded-lg p-3 text-center cursor-pointer hover:bg-secondary/50 transition-colors">
                  {uploading ? (
                    <div className="flex items-center justify-center gap-2 text-sm text-muted-foreground">
                      <Loader2 className="w-4 h-4 animate-spin" /> Uploading...
                    </div>
                  ) : form.file_url ? (
                    <p className="text-sm text-primary truncate">File uploaded ✓</p>
                  ) : (
                    <div className="flex items-center justify-center gap-2 text-sm text-muted-foreground">
                      <Upload className="w-4 h-4" /> Click to upload PDF
                    </div>
                  )}
                </div>
                <input type="file" className="hidden" onChange={handleFileUpload} accept=".pdf,.doc,.docx" />
              </label>
            </div>
          </div>
          <div>
            <Label>Key Findings</Label>
            <Textarea value={form.key_findings} onChange={e => setForm(p => ({ ...p, key_findings: e.target.value }))} placeholder="Main findings..." rows={2} />
          </div>
          <div>
            <Label>Methodology</Label>
            <Textarea value={form.methodology} onChange={e => setForm(p => ({ ...p, methodology: e.target.value }))} placeholder="Research methodology..." rows={2} />
          </div>
          <div>
            <Label>Tags (press Enter)</Label>
            <Input value={tagInput} onChange={e => setTagInput(e.target.value)} onKeyDown={handleAddTag} placeholder="Add tag..." />
            {form.tags.length > 0 && (
              <div className="flex flex-wrap gap-1.5 mt-2">
                {form.tags.map((tag, i) => (
                  <span key={i} onClick={() => setForm(p => ({ ...p, tags: p.tags.filter((_, idx) => idx !== i) }))}
                    className="px-2 py-0.5 text-xs bg-primary/10 text-primary rounded-full cursor-pointer hover:bg-destructive/10 hover:text-destructive transition-colors">
                    {tag} ×
                  </span>
                ))}
              </div>
            )}
          </div>
        </div>
        <DialogFooter>
          <Button variant="outline" onClick={() => onOpenChange(false)}>Cancel</Button>
          <Button onClick={handleSubmit} disabled={!form.title.trim() || !form.workspace_id}>
            {paper ? "Update" : "Add Paper"}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
