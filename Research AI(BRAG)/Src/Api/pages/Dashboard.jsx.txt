import React from "react";
import { base44 } from "@/api/base44Client";
import { useQuery } from "@tanstack/react-query";
import { Link } from "react-router-dom";
import { createPageUrl } from "@/utils";
import { FolderOpen, FileText, MessageSquare, TrendingUp, Plus, ArrowRight } from "lucide-react";
import { Button } from "@/components/ui/button";
import { motion } from "framer-motion";
import StatCard from "../components/dashboard/StatCard";
import RecentActivity from "../components/dashboard/RecentActivity";

export default function Dashboard() {
  const { data: workspaces = [] } = useQuery({
    queryKey: ["workspaces"],
    queryFn: () => base44.entities.Workspace.list("-created_date"),
  });

  const { data: papers = [] } = useQuery({
    queryKey: ["papers"],
    queryFn: () => base44.entities.Paper.list("-created_date"),
  });

  const activeWorkspaces = workspaces.filter(w => w.status === "active").length;

  return (
    <div className="p-6 lg:p-10 max-w-7xl mx-auto space-y-8">
      {/* Header */}
      <motion.div initial={{ opacity: 0, y: -10 }} animate={{ opacity: 1, y: 0 }}>
        <h1 className="text-2xl lg:text-3xl font-bold text-foreground tracking-tight">Dashboard</h1>
        <p className="text-sm text-muted-foreground mt-1">Overview of your research activity</p>
      </motion.div>

      {/* Stats */}
      <div className="grid grid-cols-2 lg:grid-cols-4 gap-4">
        <StatCard title="Workspaces" value={workspaces.length} icon={FolderOpen} color="indigo" subtitle={`${activeWorkspaces} active`} />
        <StatCard title="Papers" value={papers.length} icon={FileText} color="emerald" subtitle="Imported" />
        <StatCard title="Active Research" value={activeWorkspaces} icon={TrendingUp} color="amber" subtitle="In progress" />
        <StatCard title="AI Sessions" value="∞" icon={MessageSquare} color="rose" subtitle="Available" />
      </div>

      {/* Quick Actions + Recent Activity */}
      <div className="grid lg:grid-cols-3 gap-6">
        <div className="lg:col-span-1 space-y-4">
          <h3 className="text-sm font-semibold text-foreground">Quick Actions</h3>
          <div className="space-y-2">
            <Link to={createPageUrl("Workspaces")}>
              <Button className="w-full justify-start gap-3 h-11" variant="outline">
                <Plus className="w-4 h-4" /> New Workspace
              </Button>
            </Link>
            <Link to={createPageUrl("Papers")}>
              <Button className="w-full justify-start gap-3 h-11" variant="outline">
                <FileText className="w-4 h-4" /> Add Paper
              </Button>
            </Link>
            <Link to={createPageUrl("ResearchChat")}>
              <Button className="w-full justify-start gap-3 h-11 bg-primary text-primary-foreground hover:bg-primary/90">
                <MessageSquare className="w-4 h-4" /> Start AI Analysis
                <ArrowRight className="w-3 h-3 ml-auto" />
              </Button>
            </Link>
          </div>
        </div>
        <div className="lg:col-span-2">
          <RecentActivity papers={papers} workspaces={workspaces} />
        </div>
      </div>
    </div>
  );
}
