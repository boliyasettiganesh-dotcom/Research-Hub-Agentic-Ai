import React, { useState } from "react";
import { base44 } from "@/api/base44Client";
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { Link } from "react-router-dom";
import { createPageUrl } from "@/utils";
import { ArrowLeft, Plus, FileText, Trash2, Settings } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { motion } from "framer-motion";
import PaperCard from "../components/paper/PaperCard";
import PaperFormDialog from "../components/paper/PaperFormDialog";
import WorkspaceFormDialog from "../components/workspace/WorkspaceFormDialog";
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
} from "@/components/ui/alert-dialog";

export default function WorkspaceDetail() {
  const params = new URLSearchParams(window.location.search);
  const workspaceId = params.get("id");
  const queryClient = useQueryClient();

  const [showPaperForm, setShowPaperForm] = useState(false);
  const [editPaper, setEditPaper] = useState(null);
  const [showEditWs, setShowEditWs] = useState(false);
  const [showDelete, setShowDelete] = useState(false);

  const { data: workspace } = useQuery({
    queryKey: ["workspace", workspaceId],
    queryFn: async () => {
      const list = await base44.entities.Workspace.filter({ id: workspaceId });
      return list[0];
    },
    enabled: !!workspaceId,
  });

  const { data: papers = [] } = useQuery({
    queryKey: ["workspace-papers", workspaceId],
    queryFn: () => base44.entities.Paper.filter({ workspace_id: workspaceId }, "-created_date"),
    enabled: !!workspaceId,
  });

  const { data: allWorkspaces = [] } = useQuery({
    queryKey: ["workspaces"],
    queryFn: () => base44.entities.Workspace.list(),
  });

  const createPaperMutation = useMutation({
    mutationFn: (data) => base44.entities.Paper.create(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["workspace-papers", workspaceId] });
      setShowPaperForm(false);
    },
  });

  const updatePaperMutation = useMutation({
    mutationFn: ({ id, data }) => base44.entities.Paper.update(id, data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["workspace-papers", workspaceId] });
      setEditPaper(null);
      setShowPaperForm(false);
    },
  });

  const updateWsMutation = useMutation({
    mutationFn: (data) => base44.entities.Workspace.update(workspaceId, data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["workspace", workspaceId] });
      queryClient.invalidateQueries({ queryKey: ["workspaces"] });
      setShowEditWs(false);
    },
  });

  const deleteWsMutation = useMutation({
    mutationFn: () => base44.entities.Workspace.delete(workspaceId),
    onSuccess: () => {
      window.location.href = createPageUrl("Workspaces");
    },
  });

  if (!workspace) {
    return (
      <div className="p-10 text-center text-muted-foreground">Loading...</div>
    );
  }

  const statusColors = {
    active: "bg-emerald-500/10 text-emerald-600",
    completed: "bg-primary/10 text-primary",
    archived: "bg-muted text-muted-foreground",
  };

  return (
    <div className="p-6 lg:p-10 max-w-7xl mx-auto space-y-6">
      <motion.div initial={{ opacity: 0, y: -10 }} animate={{ opacity: 1, y: 0 }}>
        <Link to={createPageUrl("Workspaces")} className="inline-flex items-center gap-2 text-sm text-muted-foreground hover:text-foreground transition-colors mb-4">
          <ArrowLeft className="w-4 h-4" /> Back to Workspaces
        </Link>
        <div className="flex flex-col sm:flex-row sm:items-start justify-between gap-4">
          <div>
            <div className="flex items-center gap-3">
              <h1 className="text-2xl lg:text-3xl font-bold text-foreground tracking-tight">{workspace.name}</h1>
              <Badge className={statusColors[workspace.status]}>{workspace.status}</Badge>
            </div>
            {workspace.description && (
              <p className="text-sm text-muted-foreground mt-2">{workspace.description}</p>
            )}
            {workspace.tags?.length > 0 && (
              <div className="flex flex-wrap gap-1.5 mt-3">
                {workspace.tags.map((tag, i) => (
                  <Badge key={i} variant="outline" className="text-xs">{tag}</Badge>
                ))}
              </div>
            )}
          </div>
          <div className="flex gap-2">
            <Button variant="outline" size="sm" onClick={() => setShowEditWs(true)} className="gap-1.5">
              <Settings className="w-3.5 h-3.5" /> Edit
            </Button>
            <Button variant="outline" size="sm" onClick={() => setShowDelete(true)} className="gap-1.5 text-destructive hover:text-destructive">
              <Trash2 className="w-3.5 h-3.5" /> Delete
            </Button>
          </div>
        </div>
      </motion.div>

      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <FileText className="w-4 h-4 text-muted-foreground" />
          <span className="text-sm font-medium text-foreground">{papers.length} Papers</span>
        </div>
        <Button size="sm" onClick={() => { setEditPaper(null); setShowPaperForm(true); }} className="gap-1.5">
          <Plus className="w-3.5 h-3.5" /> Add Paper
        </Button>
      </div>

      <div className="grid gap-4">
        {papers.map((paper, i) => (
          <PaperCard key={paper.id} paper={paper} index={i} onEdit={(p) => { setEditPaper(p); setShowPaperForm(true); }} />
        ))}
      </div>

      {papers.length === 0 && (
        <div className="text-center py-16 bg-card rounded-xl border border-border">
          <FileText className="w-10 h-10 text-muted-foreground mx-auto mb-3" />
          <p className="text-muted-foreground">No papers in this workspace yet</p>
          <Button size="sm" className="mt-4 gap-1.5" onClick={() => setShowPaperForm(true)}>
            <Plus className="w-3.5 h-3.5" /> Add First Paper
          </Button>
        </div>
      )}

      <PaperFormDialog
        open={showPaperForm}
        onOpenChange={(v) => { setShowPaperForm(v); if (!v) setEditPaper(null); }}
        onSubmit={(data) => editPaper ? updatePaperMutation.mutate({ id: editPaper.id, data }) : createPaperMutation.mutate(data)}
        paper={editPaper}
        workspaces={allWorkspaces}
        defaultWorkspaceId={workspaceId}
      />

      <WorkspaceFormDialog
        open={showEditWs}
        onOpenChange={setShowEditWs}
        onSubmit={(data) => updateWsMutation.mutate(data)}
        workspace={workspace}
      />

      <AlertDialog open={showDelete} onOpenChange={setShowDelete}>
        <AlertDialogContent>
          <AlertDialogHeader>
            <AlertDialogTitle>Delete Workspace</AlertDialogTitle>
            <AlertDialogDescription>
              This will permanently delete "{workspace.name}". Papers in this workspace will not be deleted.
            </AlertDialogDescription>
          </AlertDialogHeader>
          <AlertDialogFooter>
            <AlertDialogCancel>Cancel</AlertDialogCancel>
            <AlertDialogAction onClick={() => deleteWsMutation.mutate()} className="bg-destructive text-destructive-foreground hover:bg-destructive/90">
              Delete
            </AlertDialogAction>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>
    </div>
  );
}
